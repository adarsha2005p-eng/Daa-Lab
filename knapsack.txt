#include<stdio.h>

int main()
{
    int i,n;
    float w[10],p[10],ratio[10],cap,profit=0;

    printf("Enter number of items: ");
    scanf("%d",&n);

    printf("Enter weight and profit:\n");
    for(i=0;i<n;i++)
    {
        scanf("%f %f",&w[i],&p[i]);
        ratio[i] = p[i]/w[i];
    }

    printf("Enter capacity: ");
    scanf("%f",&cap);

    for(i=0;i<n;i++)
    {
        if(w[i] <= cap)
        {
            profit += p[i];
            cap -= w[i];
        }
        else
        {
            profit += p[i]*(cap/w[i]);
            break;
        }
    }

    printf("Maximum Profit = %.2f",profit);
}
